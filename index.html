x64 Assembly: registers

  • Registers: a, b, c, d, si, di, sp, bp, r8-r15, xmm0-xmm15
  • Register widths: 8: ah/al, 16: ax, 32: eax, 64: rax
  • Function params: rdi, rsi, rdx, rcx, r8, r9
  • Function return value in: rax, rdx, xmm0

x64 Assembly: instructions

Intel syntax 
		oper
		oper dest
		oper dest src
		oper dest [src ptr]

Array bounds - optimization

C++asm

	int fun(int i)
	{
		int array[4];
		array[i] = 333;
		return array[i];
	}


	fun(int):
	mov eax, 333
	ret
Params: di, si, d, c, r8, r9
Returns: a, d, xmm0

Array bounds - optimization

C++asm

					int fun(int i, int x)
					{
						int array[4];
						array[i] = x;
						return array[i];
					}
			

				fun(int, int):
				mov eax, esi
				ret
Params: di, si, d, c, r8, r9
Returns: a, d, xmm0

Array bounds - the 5th element

C++asm

					bool is_in_arr(int val)
					{
						int array[] = {1, 2, 3, 5};
					
						for (int i = 0; i <= 4; i++)
							if (array[i] == val)
								return true;
						
						return false;
					}
			

					is_in_arr(int):
					  mov eax, 1
					  ret
Params: di, si, d, c, r8, r9
Returns: a, d, xmm0

Integer overflow - optimization

C++asm

					int foo(int a, int b)
					{
					  for(int i = 0; i < 10; i++)
					  {
						if (b+i > b)
							return 6;
						a++;
					  }
					  return a;
					}
			

					foo(int, int):
					  mov eax, 6
					  ret
Params: di, si, d, c, r8, r9
Returns: a, d, xmm0

Integer overflow - optimization

C++asm

					int foo(int a, unsigned b)
					{
					  for(int i = 0; i < 10; i++)
					  {
						if (b+i > b)
							return 6;
						a++;
					  }
					  return a;
					}
			

foo(int, unsigned int):
	mov eax, edi
	lea ecx, [rdi+10]
	jmp .L2
.L4:
	lea edx, [rax+rsi]
	sub edx, edi
	cmp esi, edx
	jb .L5
.L2:
	add eax, 1
	cmp eax, ecx
	jne .L4
	rep ret
.L5:
	mov eax, 6
	ret
Params: di, si, d, c, r8, r9
Returns: a, d, xmm0
